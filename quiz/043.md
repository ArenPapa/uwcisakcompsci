class palNum:
    def __init__(self, A, B):
        self.A = A
        self.B = B
    
    def get_pal_list(self):
        return [num for num in range(self.A, self.B + 1) if str(num) == str(num)[::-1]]

# Example usage:
pal1 = palNum(1, 9)
print(pal1.get_pal_list())  

pal2 = palNum(10, 199)
print(pal2.get_pal_list())
