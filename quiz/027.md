    def sort_dict_by_values(input_dict):
        return dict(sorted(input_dict.items(), key=lambda item: item[1]))
    
    
    example_dict = {'a': 3, 'b': 1, 'c': 2}
    sorted_dict = sort_dict_by_values(example_dict)
    print(sorted_dict)  # Output: {'b': 1, 'c': 2, 'a': 3}
