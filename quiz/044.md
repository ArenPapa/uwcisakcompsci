    class rainDrops:
        def pour(self, n: int) -> str:
            result = ""
            if n % 3 == 0:
                result += "Pling"
            if n % 5 == 0:
                result += "Plang"
            if n % 7 == 0:
                result += "Plong"
            
            return result if result else str(n)
    
    # Example usage:
    rain = rainDrops()
    print(rain.pour(28))  # Output: "Plong"
    print(rain.pour(30))  # Output: "PlingPlang"
    print(rain.pour(34))  # Output: "34"
